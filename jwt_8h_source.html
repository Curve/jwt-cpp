542 template<typename error_category = error::rsa_error>
543 std::string extract_pubkey_from_cert(const std::string& certstr, const std::string& pw, std::error_code& ec) {
544 ec.clear();
545 auto certbio = make_mem_buf_bio(certstr);
546 auto keybio = make_mem_buf_bio();
547 if (!certbio || !keybio) {
548 ec = error_category::create_mem_bio_failed;
549 return {};
550 }
551
552 std::unique_ptr<X509, decltype(&X509_free)> cert(
553 PEM_read_bio_X509(certbio.get(), nullptr, nullptr, const_cast<char*>(pw.c_str())), X509_free);
554 if (!cert) {
555 ec = error_category::cert_load_failed;
556 return {};
557 }
558 std::unique_ptr<EVP_PKEY, decltype(&EVP_PKEY_free)> key(X509_get_pubkey(cert.get()), EVP_PKEY_free);
559 if (!key) {
560 ec = error_category::get_key_failed;
561 return {};
562 }
563 if (PEM_write_bio_PUBKEY(keybio.get(), key.get()) == 0) {
564 ec = error_category::write_key_failed;
565 return {};
566 }
567
568 return write_bio_to_string<error_category>(keybio, ec);
569 }

714 template<typename error_category = error::rsa_error>
715 evp_pkey_handle load_public_key_from_string(const std::string& key, const std::string& password,
716 std::error_code& ec) {
717 ec.clear();
718 auto pubkey_bio = make_mem_buf_bio();
719 if (!pubkey_bio) {
720 ec = error_category::create_mem_bio_failed;
721 return {};
722 }
723 if (key.substr(0, 27) == "-----BEGIN CERTIFICATE-----") {
724 auto epkey = helper::extract_pubkey_from_cert<error_category>(key, password, ec);
725 if (ec) return {};
726 const int len = static_cast<int>(epkey.size());
727 if (BIO_write(pubkey_bio.get(), epkey.data(), len) != len) {
728 ec = error_category::load_key_bio_write;
729 return {};
730 }
731 } else {
732 const int len = static_cast<int>(key.size());
733 if (BIO_write(pubkey_bio.get(), key.data(), len) != len) {
734 ec = error_category::load_key_bio_write;
735 return {};
736 }
737 }
738
739 evp_pkey_handle pkey(PEM_read_bio_PUBKEY(
740 pubkey_bio.get(), nullptr, nullptr,
741 (void*)password.data())); // NOLINT(google-readability-casting) requires `const_cast`
742 if (!pkey) ec = error_category::load_key_bio_read;
743 return pkey;
744 }

1689 std::string der_to_p1363_signature(const std::string& der_signature, std::error_code& ec) const {
1690 const unsigned char* possl_signature = reinterpret_cast<const unsigned char*>(der_signature.data());
1691 std::unique_ptr<ECDSA_SIG, decltype(&ECDSA_SIG_free)> sig(
1692 d2i_ECDSA_SIG(nullptr, &possl_signature, static_cast<long>(der_signature.length())),
1693 ECDSA_SIG_free);
1694 if (!sig) {
1695 ec = error::signature_generation_error::signature_decoding_failed;
1696 return {};
1697 }
1698
1699#ifdef JWT_OPENSSL_1_0_0
1700 auto rr = helper::bn2raw(sig->r);
1701 auto rs = helper::bn2raw(sig->s);
1702#else
1703 const BIGNUM* r;
1704 const BIGNUM* s;
1705 ECDSA_SIG_get0(sig.get(), &r, &s);
1706 auto rr = helper::bn2raw(r);
1707 auto rs = helper::bn2raw(s);
1708#endif
1709 if (rr.size() > signature_length / 2 || rs.size() > signature_length / 2)
1710 throw std::logic_error("bignum size exceeded expected length");
1711 rr.insert(0, signature_length / 2 - rr.size(), '\0');
1712 rs.insert(0, signature_length / 2 - rs.size(), '\0');
1713 return rr + rs;
1714 }

1716 std::string p1363_to_der_signature(const std::string& signature, std::error_code& ec) const {
1717 ec.clear();
1718 auto r = helper::raw2bn(signature.substr(0, signature.size() / 2), ec);
1719 if (ec) return {};
1720 auto s = helper::raw2bn(signature.substr(signature.size() / 2), ec);
1721 if (ec) return {};
1722
1723 ECDSA_SIG* psig;
1724#ifdef JWT_OPENSSL_1_0_0
1725 ECDSA_SIG sig;
1726 sig.r = r.get();
1727 sig.s = s.get();
1728 psig = &sig;
1729#else
1730 std::unique_ptr<ECDSA_SIG, decltype(&ECDSA_SIG_free)> sig(ECDSA_SIG_new(), ECDSA_SIG_free);
1731 if (!sig) {
1732 ec = error::signature_verification_error::create_context_failed;
1733 return {};
1734 }
1735 ECDSA_SIG_set0(sig.get(), r.release(), s.release());
1736 psig = sig.get();
1737#endif
1738
1739 int length = i2d_ECDSA_SIG(psig, nullptr);
1740 if (length < 0) {
1741 ec = error::signature_verification_error::signature_encoding_failed;
1742 return {};
1743 }
1744 std::string der_signature(length, '\0');
1745 unsigned char* psbuffer = (unsigned char*)der_signature.data();
1746 length = i2d_ECDSA_SIG(psig, &psbuffer);
1747 if (length < 0) {
1748 ec = error::signature_verification_error::signature_encoding_failed;
1749 return {};
1750 }
1751 der_signature.resize(length);
1752 return der_signature;
1753 }

3553 std::mbstate_t state = std::mbstate_t();
3554 const char* in_next = str.data();
3555 const char* in_end = str.data() + str.size();
3556 std::wstring wide;
3557 wide.reserve(str.size());
3558
3559 while (in_next != in_end) {
3560 wchar_t wc;
3561 std::size_t result = std::mbrtowc(&wc, in_next, in_end - in_next, &state);
3562 if (result == static_cast<std::size_t>(-1)) {
3563 throw std::runtime_error("encoding error: " + std::string(std::strerror(errno)));
3564 } else if (result == static_cast<std::size_t>(-2)) {
3565 throw std::runtime_error("conversion error: next bytes constitute an incomplete, but so far "
3566 "valid, multibyte character.");
3567 }
3568 in_next += result;
3569 wide.push_back(wc);
3570 }
3571
3572 auto& f = std::use_facet<std::ctype<wchar_t>>(loc);
3573 f.tolower(&wide[0], &wide[0] + wide.size());
3574
3575 std::string out;
3576 out.reserve(wide.size());
3577 for (wchar_t wc : wide) {
3578 char mb[MB_LEN_MAX];
3579 std::size_t n = std::wcrtomb(mb, wc, &state);
3580 if (n != static_cast<std::size_t>(-1)) out.append(mb, n);
3581 }
